139void lcd_set_pos(int line, int pos)
140{
141 /*Display auf Home Position*/
142 waitForBusyLCD();
143 while( PMMODEbits.BUSY );
144 PMADDR = 0;
145 PMDIN1 = LCD_DISPLAY_HOME;
146
147 /*Berechnung wieivel geshiftet werden muss*/
148
149 int i = 0;
150 int to_shift = 0;
151
152 if (line == 1) //Erste Linie
153 {
154 to_shift = pos;
155 }
156
157 else //Zweite Linie
158 {
159 to_shift = pos + 40;
160 }
161
162 for(i = 0; i < to_shift - 1; i++)
163 {
164 waitForBusyLCD();
165 while( PMMODEbits.BUSY );
166 PMADDR = 0;
167 PMDIN1 = CURSOR_OR_DISPLAY;
