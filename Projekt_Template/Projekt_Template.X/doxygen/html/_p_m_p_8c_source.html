29void initPMP(void)
30{
31 //Alle Ausgänge als Digital
32 _ANSB15 = 0;
33 ANSELE &= 0xFF00;
34
35 //CON
36 PMCONbits.PMPEN = 1;
37 PMCONbits.PSIDL = 0;
38 PMCONbits.ADRMUX = 0b00;
39 PMCONbits.PTBEEN = 0;
40 PMCONbits.PTWREN = 1;
41 PMCONbits.PTRDEN = 1;
42 PMCONbits.CSF = 0b00;
43 PMCONbits.ALP = 1;
44 PMCONbits.CS2P = 1;
45 PMCONbits.CS1P = 1;
46 PMCONbits.BEP = 1;
47 PMCONbits.WRSP = 1;
48 PMCONbits.RDSP = 1;
49
50 //MODE
51 PMMODEbits.IRQM = 0b00;
52 PMMODEbits.INCM = 0b00;
53 PMMODEbits.MODE16 = 0;
54 PMMODEbits.MODE = 0b11; //Master Mode 1
55 PMMODEbits.WAITB = 0b11;
56 PMMODEbits.WAITM = 0b1111;
57 PMMODEbits.WAITE = 0b11;
58
59
60 //ADRESS ENABLE
61
62 /* Parallel Master Port Address Enable Register (PMAEN). Dieses Register
63 * steuert den Betrieb der Adress- und Chip-Select-Pins, die mit dem PMP
64 * Modul zugeordnet sind.*/
65 PMAEN = 0x0001; // PMA0 enabled
66
67 /* Pad Configuration Control Register (PADCFG1). Das PADCFG1 steuert,
68 * welcher digitale Eingangspuffer vom PMP-Modul verwendet wird.
69 * Das PMPTTL Bit ermöglicht dem Benutzer die Auswahl zwischen
70 * Transistor-Transistor-Logik (TTL) und Schmitt-Trigger
71 * (ST)-Eingangspuffern für eine bessere Kompatibilität mit externen
72 * Schaltungen. PMPTTL = 0 ist die Standardkonfiguration und wählt die
73 * ST-Puffer aus.*/
74 PADCFG1bits.PMPTTL = 0;
75 delay_ms(40); //40 ms warten
76
77
79 /* Parallel Master Port Address Register (PMADDR) Dieses Register enthält
80 * die Adresse, an die die ausgehenden Daten geschrieben werden sollen,
81 * sowie die Chip Select Steuerbits für die Adressierung von parallelen
82 * Slave-Geräten.*/
83 PMADDR = 0; //RS auf 0
84
85 /* Der Register Parallel Master Port Data Input 1 (PMDIN1) dient der
86 * Pufferung eingehender Daten.*/
87 PMDIN1 = LCD_CMD_INIT; //Function Set 1
88 __delay_us(4100); //4.1 ms warten
89
90 PMDIN1 = LCD_CMD_INIT; //Function Set 2
91 __delay_us(100); //100 us warten
92
93 PMDIN1 = LCD_CMD_INIT; //Function Set 3
94 __delay_us(38); //LCD_CMD_INIT benötigt 38 us zum Ausführen
95
96
98 /*LCD_FUNCTION_SET stellt die Bewegungsrichtig von Cursor und Display ein.*/
99 PMDIN1 = LCD_FUNCTION_SET;
100 __delay_us(38); //LCD_FUNCTION_SET benötigt 38 us zum Ausführen
101
102 /*LCD_DISPLAY_OFF schaltet das Display aus.*/
103 PMDIN1 = LCD_DISPLAY_OFF;
104 __delay_us(38); //LCD_DISPLAY_OFF benötigt 38 us zum Ausführen
105
106 /*LCD_DISPLAY_CLEAR löscht die Anzeigedaten. LCD_DISPLAY_CLEAR benötigt
107 * 1520 us zum Ausführen*/
108 PMDIN1 = LCD_DISPLAY_CLEAR;
109 __delay_us(1520);
110
111 /*LCD_ENTRY_MODE stellt den Eingabemodus ein. Der Cursor bewegt sich dabei
112 *nach rechts.*/
113 PMDIN1 = LCD_ENTRY_MODE;
114 __delay_us(38); //LCD_ENTRY_MODE benötigt 38 us zum Ausführen
115
116 /*LCD_DISPLAY_ON schaltet das Display ein.*/
117 PMDIN1 = LCD_DISPLAY_ON;
118 __delay_us(38); //LCD_DISPLAY_ON benötigt 38 us zum Ausführen
119
120}/*initPMP()*/

126uint8_t lcd_get_status(void)
127{
128 uint8_t dummy;
129
130 /* Parallel Master Port Mode Register (PMMODE). Dieses Register wählt den
131 * Master/Slave-Modus aus und konfiguriert die Betriebsmodi des PMP-Moduls.
132 * Dieses Register enthält das universelle Status-Flag BUSY, das in
133 * Master-Modi verwendet wird, um anzuzeigen, dass eine Operation durch das
134 * Modul im Gange ist*/
135 while( PMMODEbits.BUSY); //Warten bis PMP bereit
136 PMADDR = 0;
137
138 dummy = PMDIN1; //lesen anstoßen durch dummy read
139
140 while( PMMODEbits.BUSY);
141 return (PMDIN1);
142
143}/*lcd_get_status()*/

216void lcd_set_pos(int line, int pos)
217{
218 /*Display auf Home Position*/
219 waitForBusyLCD();
220 while( PMMODEbits.BUSY );
221 PMADDR = 0;
222
223 /* LCD_DISPLAY_HOME setzt Cursor und Anzeige an ihren ursprünglichen Zustand
224 * zurück.*/
225 PMDIN1 = LCD_DISPLAY_HOME;
226
227 /*Berechnung wieivel geshiftet werden muss*/
228 int i = 0;
229 int to_shift = 0;
230
231 if (line == 1) //Erste Linie
232 {
233 to_shift = pos;
234 }
235
236 else //Zweite Linie
237 {
238 to_shift = pos + 40;
239 }
240
241 for(i = 0; i < to_shift - 1; i++)
242 {
243 waitForBusyLCD();
244 while( PMMODEbits.BUSY );
245 PMADDR = 0;
246
247 /*CURSOR_OR_DISPLAY schiebt Cursor und Anzeige nach rechts (der Cursor
248 *bewegt sich entsprechend der Anzeige).*/
249 PMDIN1 = CURSOR_OR_DISPLAY;
250 }
251
252}/*lcd_set_pos()*/
