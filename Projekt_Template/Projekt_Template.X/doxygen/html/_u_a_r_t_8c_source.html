82int16_t putcFIFO_TX(char c)
83{
84 _GIE=0;
85 //if (buffer.write >= BUFFER_SIZE)

88 if ( ( FIFO.write + 1 == FIFO.read ) ||
89 ( FIFO.read == 0 && FIFO.write + 1 == BUFFER_SIZE ) )
90 {
91 _GIE=1;
92 return BUFFER_FAIL; // voll
93 }
94
95 FIFO.data[FIFO.write] = c;

97 FIFO.write++;
