40void ConfigureOscillator(void)
41{
42 if (SYS_FREQ>7370000L) //Nur umschalten auf Primary (8 MHz) wenn höhere Frequenz erwünscht
43 {
44 switch (SYS_FREQ)
45 {
46 case 8000000L:
47 //PLL muss nicht konfiguriert werden
48 // externer Quartz mit 8Mhz
49 break;
50 case 50000000L:
51 CLKDIVbits.PLLPOST=2; //N2=4
52 PLLFBD=48; //M=50
53 CLKDIVbits.PLLPRE=0; //N1=2
54 break;
55 case 70000000L:
56 CLKDIVbits.PLLPOST=2; //N2=4
57 PLLFBD=188; //M=190
58 CLKDIVbits.PLLPRE=3; //N1=5
59 break;
60 case 100000000L:
61 CLKDIVbits.PLLPOST=0; //N2=2
62 PLLFBD=123; //M=125
63 CLKDIVbits.PLLPRE=3; //N1=5
64 break;
65 case 140000000L:
66 CLKDIVbits.PLLPOST=0; //N2=2
67 PLLFBD=173; //M=175
68 CLKDIVbits.PLLPRE=3; //N1=5
69 break;
70 //default:
71 //#error Tets
72 }
73 OSCTUN = 0;
74
75 if (SYS_FREQ == 8000000L)
76 {
77 __builtin_write_OSCCONH(0x02); //Switch auf Primary ohne PLL
78
79 __builtin_write_OSCCONL(OSCCON | 0x01);
80 while (OSCCONbits.COSC!= 0x02); //Warten bis gewechselt wurde
81 }
82
83 else
84 {
85 __builtin_write_OSCCONH(0x03); //Switch auf Primary mit PLL
86
87 __builtin_write_OSCCONL(OSCCON | 0x01);
88
89 while (OSCCONbits.COSC!= 0x3); //Warten bis gewechselt wurde
90 while (OSCCONbits.LOCK!= 1);
91 }
92
93 }
94}

98void init_timer1() //generiert in 1s Rythmus Interrupts
99{
100 __builtin_write_OSCCONL(0b00000011); //SOSC aktivieren
101 T1CONbits.TON = 0; // Disable Timer
102 T1CONbits.TCS = 1; // Select external clock
103 T1CONbits.TSYNC = 0; // Disable Synchronization
104 T1CONbits.TCKPS = 0b00; // Select 1:1 Prescaler
105 TMR1 = 0x00; // Clear timer register
106 PR1 = 32767; // Load the period value, Quarztakt
107
108 IPC0bits.T1IP = 2; // Set Timer 1 Interrupt Priority Level
109 IFS0bits.T1IF = 0; // Clear Timer 1 Interrupt Flag
110 IEC0bits.T1IE = 1; // Enable Timer1 interrupt
111 T1CONbits.TON = 1; // Start Timer
112}
