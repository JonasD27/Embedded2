175void lcd_set_pos(int line, int pos)
176{
177 waitForBusyLCD();
178 TRISE &= 0xFF00; //Datenbus als Ausgang
179
180 //Position auf 0,0 setzten
181 LCD_RS = 0;
182 LCD_R_W = 0;
183 __delay_cycles(4); //80 ns warten
184
185 /* LCD_DISPLAY_HOME setzt Cursor und Anzeige an ihren ursprünglichen Zustand
186 * zurück.*/
187 LCD_DATA(LCD_DISPLAY_HOME);
188 LCD_ENABLE = 1;
189 __delay_cycles(33); //660 ns warten
190 LCD_ENABLE = 0;
191
192
193 int i = 0;
194 int to_shift = 0;
195
196 if (line == 1) //Erste Linie
197 {
198 to_shift = pos;
199 }
200
201 else //Zweite Linie
202 {
203 to_shift = pos + 40;
204 }
205
206 for(i = 0; i < to_shift - 1; i++)
207 {
208 waitForBusyLCD();
209 TRISE &= 0xFF00; //Datenbus als Ausgang
210
211 LCD_RS = 0;
212 LCD_R_W = 0;
213 __delay_cycles(4); //80 ns warten
214
215 /*CURSOR_OR_DISPLAY schiebt Cursor und Anzeige nach rechts (der Cursor
216 *bewegt sich entsprechend der Anzeige).*/
217 LCD_DATA(CURSOR_OR_DISPLAY);
218 LCD_ENABLE = 1;
219 __delay_cycles(33); //660 ns warten
220 LCD_ENABLE = 0;
221 }
222
223}/*lcd_set_pos()*/

230uint8_t lcd_get_status(void)
231{
232 uint8_t received_data;
233 TRISE = TRISE | 0x00FF; //Datenbus als Eingang
234
235 LCD_RS = 0;
236 LCD_R_W = 1;
237 __delay_cycles(4); //80 ns warten
238 LCD_ENABLE = 1;
239 __delay_cycles(20); //min 360 ns warten -> 400ns
240 received_data = PORTE;
241 LCD_ENABLE = 0;
242
243 return received_data;
244
245}/*lcd_get_status()*/
