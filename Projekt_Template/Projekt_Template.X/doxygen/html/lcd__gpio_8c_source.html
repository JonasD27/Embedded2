177void lcd_set_pos(int line, int pos)
178{
179 waitForBusyLCD();
180
181 TRISE &= 0xFF00; //Datenbus als Ausgang
182
183 //Position auf 0,0 setzten
184 LCD_RS = 0;
185 LCD_R_W = 0;
186 __delay_cycles(4);
187 LCD_DATA(LCD_DISPLAY_HOME);
188 LCD_ENABLE = 1;
189 __delay_cycles(33);
190 LCD_ENABLE = 0;
191 //__delay_us(1520); //LCD_DISPLAY_HOME benötigt 1.52 ms zum Ausführen
192
193 int i = 0;
194 int to_shift = 0;
195
196 if (line == 1) //Erste Linie
197 {
198 to_shift = pos;
199 }
200
201 else //Zweite Linie
202 {
203 to_shift = pos + 40;
204 }
205
206 for(i = 0; i < to_shift - 1; i++)
207 {
208 waitForBusyLCD();
209 TRISE &= 0xFF00; //Datenbus als Ausgang
210 LCD_RS = 0;
211 LCD_R_W = 0;
212 __delay_cycles(4);
213 LCD_DATA(CURSOR_OR_DISPLAY);
214 LCD_ENABLE = 1;
215 __delay_cycles(33);
216 LCD_ENABLE = 0;
217 //__delay_us(38); //CURSOR_OR_DISPLAY benötigt 38 us zum Ausführen
218 }
219
220}/*lcd_set_pos()*/
