19int16_t put_I2C_struct_FIFO(I2C_struct s)
20{
21 _GIE=0;
22 if ( ( FIFO_I2C.write + 1 == FIFO_I2C.read ) ||
23 ( FIFO_I2C.read == 0 && FIFO_I2C.write + 1 == BUFFER_SIZE ) )
24 {
25 _GIE=1;
26 return BUFFER_FAIL; // voll
27 }
28
29 FIFO_I2C.data[FIFO_I2C.write] = s;
30
31 FIFO_I2C.write++;
32 if (FIFO_I2C.write >= BUFFER_SIZE)
33 {
34 FIFO_I2C.write = 0;
35 }
36 _GIE=1;
37 return BUFFER_SUCCESS;

39} /* put_I2C_struct_FIFO() */

112void initI2C()
113{
114 I2C2CONbits.A10M = 0;

211void *FSM_Adresse_Write(void)
212{
213 if(I2C2STATbits.TRSTAT==0) //Wenn erfolgreich übertragen
214 {
215 if (I2C2STATbits.ACKSTAT==1) // Leitet Stop-Bedigungen weiter
216 {
217 I2C2CONbits.PEN=1; //Fehler bei Kommunikation
218 *I2C_test_struct.status=Error;
219 return FSM_Stop;
220 }
221
222 if (I2C2STATbits.ACKSTAT==0) //Wenn ACK von Slave erhalten
223 {
224 static int count=0;
225
226 if (count < I2C_test_struct.num_write) //Noch Bytes zu senden
227 {
228
229 I2C2TRN=I2C_test_struct.writebuf[count];
230 count++;
231 return FSM_Adresse_Write;
232 }
233
234 else //Nichts mehr zu schicken
235 {
236 count=0;
237 I2C2CONbits.RSEN=1; // Leitet Bedingungen für den Restart weiter
238 return FSM_Repeated_Start;
239 }
240
241 }
242
243 }
244 return FSM_Adresse_Write;

276void *FSM_Adresse_Read(void)
277{
278 if(I2C2STATbits.TRSTAT==0) //Wenn erfolgreich übertragen
279 {
280 if (I2C2STATbits.ACKSTAT==1) //Wenn NACK von Slave erhalten
281 {
282 I2C2CONbits.PEN=1; // Leitet Stop-Bedigungen weiter
283 *I2C_test_struct.status=Error; //Fehler bei Kommunikation
284 return FSM_Stop;
285 }
286
287 if (I2C2STATbits.ACKSTAT==0) //Wenn ACK von Slave erhalten
288 {
289 if (I2C2CONbits.ACKEN==0) //Wenn Bit der ACK-Sequenz freigegeben
290 {
291 static int count = 0;
292
293 if (count < I2C_test_struct.num_read) //Noch Bytes zu empfangen
294 {
295 count++;
296 I2C2CONbits.RCEN=1; // Aktiviert den Empfangsmodus für I2C
297 return FSM_RECV_EN;
298 }
299
300 else //Nichts mehr zu empfangen
301 {
302 count = 0;
303 I2C2CONbits.PEN=1; // Leitet Stop-Bedigungen weiter
304 *I2C_test_struct.status=Finished; //Anforderung abgearbeitet
305 return FSM_Stop;
306 }
307 }
308 else
309 {
310 return FSM_Adresse_Read;
311 }
312
313 }
314
315 }
316 return FSM_Adresse_Read;
317
318} /* *FSM_Adresse_Read() */

327void *FSM_RECV_EN(void)
328{
329 if (I2C2CONbits.RCEN==0) //Wenn der Empfangsmodus aktiviert wurde
330 {
331 static int count = 0;
332 I2C_test_struct.readbuf[count]=I2C2RCV;
333 count++;
334
335 if (count>=I2C_test_struct.num_read) //Wenn letztes Byte empfangen wurde
336 {
337 count=0;
338 I2C2CONbits.ACKDT=1; //Sendet einen NACK während eines Acknowledge
339 }
340 else
341 {
342 I2C2CONbits.ACKDT=0; //Sendet einen ACK während eines Acknowledge
343 }
344
345 I2C2CONbits.ACKEN=1; //Initiiert die Acknowledge-Sequenz
346 return FSM_Adresse_Read;
347 }
348
349 return FSM_RECV_EN;
350
351} /* *FSM_RECV_EN() */
