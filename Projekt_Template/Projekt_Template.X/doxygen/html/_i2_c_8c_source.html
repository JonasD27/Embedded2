30int16_t put_I2C_struct_FIFO(I2C_struct s)
31{
32 _GIE=0;
33 if ( ( FIFO_I2C.write + 1 == FIFO_I2C.read ) ||
34 ( FIFO_I2C.read == 0 && FIFO_I2C.write + 1 == BUFFER_SIZE ) )
35 {
36 _GIE=1;
37 return BUFFER_FAIL; // voll
38 }
39
40 FIFO_I2C.data[FIFO_I2C.write] = s;
41
42 FIFO_I2C.write++;
43 if (FIFO_I2C.write >= BUFFER_SIZE)
44 {
45 FIFO_I2C.write = 0;
46 }
47 _GIE=1;
48 return BUFFER_SUCCESS;

50} /* put_I2C_struct_FIFO() */

225void *FSM_Adresse_Write(void)
226{
227 if(I2C2STATbits.TRSTAT==0) //Wenn erfolgreich übertragen
228 {
229 if (I2C2STATbits.ACKSTAT==1) // Leitet Stop-Bedigungen weiter
230 {
231 I2C2CONbits.PEN=1; //Fehler bei Kommunikation
232 *I2C_test_struct.status=Error;
233 if (I2C_test_struct.callback != NULL)
234 {
235 I2C_test_struct.callback(I2C_test_struct.readbuf, I2C_test_struct.num_read, I2C_test_struct.status, I2C_test_struct.ID);
236 }
237 return FSM_Stop;
238 }
239
240 if (I2C2STATbits.ACKSTAT==0) //Wenn ACK von Slave erhalten
241 {
242 static int count=0;
243
244 if (count < I2C_test_struct.num_write) //Noch Bytes zu senden
245 {
246
247 I2C2TRN=I2C_test_struct.writebuf[count];
248 count++;
249 return FSM_Adresse_Write;
250 }
251
252 else //Nichts mehr zu schicken
253 {
254 count=0;
255 I2C2CONbits.RSEN=1; // Leitet Bedingungen für den Restart weiter
256 return FSM_Repeated_Start;
257 }
258
259 }
260
261 }
262 return FSM_Adresse_Write;

294void *FSM_Adresse_Read(void)
295{
296 if(I2C2STATbits.TRSTAT==0) //Wenn erfolgreich übertragen
297 {
298 if (I2C2STATbits.ACKSTAT==1) //Wenn NACK von Slave erhalten
299 {
300 I2C2CONbits.PEN=1; // Leitet Stop-Bedigungen weiter
301 *I2C_test_struct.status=Error; //Fehler bei Kommunikation
302 if (I2C_test_struct.callback != NULL)
303 {
304 I2C_test_struct.callback(I2C_test_struct.readbuf, I2C_test_struct.num_read, I2C_test_struct.status, I2C_test_struct.ID);
305 }
306 return FSM_Stop;
307 }
308
309 if (I2C2STATbits.ACKSTAT==0) //Wenn ACK von Slave erhalten
310 {
311 if (I2C2CONbits.ACKEN==0) //Wenn Bit der ACK-Sequenz freigegeben
312 {
313 static int count = 0;
314
315 if (count < I2C_test_struct.num_read) //Noch Bytes zu empfangen

356void *FSM_RECV_EN(void)
