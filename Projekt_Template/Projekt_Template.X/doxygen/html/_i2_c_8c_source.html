27int16_t put_I2C_struct_FIFO(I2C_struct s)
28{
29 _GIE=0;
30 if ( ( FIFO_I2C.write + 1 == FIFO_I2C.read ) ||
31 ( FIFO_I2C.read == 0 && FIFO_I2C.write + 1 == BUFFER_SIZE ) )
32 {
33 _GIE=1;
34 return BUFFER_FAIL; // voll
35 }
36
37 FIFO_I2C.data[FIFO_I2C.write] = s;
38
39 FIFO_I2C.write++;
40 if (FIFO_I2C.write >= BUFFER_SIZE)
41 {
42 FIFO_I2C.write = 0;
43 }
44 _GIE=1;
45 return BUFFER_SUCCESS;
46
47} /* put_I2C_struct_FIFO() */

122void initI2C()
123{
124 I2C2CONbits.A10M = 0;
125 I2C2BRG = 245; //100kHz
126
133 I2C_SDA_TRIS = 1;
134 I2C_SCL_TRIS = 1;
135 I2C_SDA = 0;
136 I2C_SCL = 0;
137
138 int j;
139 for (j=0; j<=9; j++) // takten bis min 1 Byte
140 {
141 I2C_SCL_TRIS = 0; delay_ms(1); // 5 us wÃ¤ren ausreichend ...100 kBaud
142 I2C_SCL_TRIS = 1; delay_ms(1);
143 }
144 // Start Condition senden
145 I2C_SCL_TRIS = 0; delay_ms(1);
146 I2C_SDA_TRIS = 0; delay_ms(1);
147 // Stop Condition senden
148 I2C_SCL_TRIS = 1; delay_ms(1);
149 I2C_SDA_TRIS = 1; delay_ms(1);
150
151 // Nun I2C erst anschalten
152 _MI2C2IF = 0; //Interrupt falls noetig
153 _MI2C2IE = 0;
154
159 I2C2CONbits.I2CEN = 1;

161} /* initI2C() */

221void *FSM_Adresse_Write(void)
222{
223 if(I2C2STATbits.TRSTAT==0) //Wenn erfolgreich übertragen
224 {
225 if (I2C2STATbits.ACKSTAT==1) // Leitet Stop-Bedigungen weiter
226 {
227 I2C2CONbits.PEN=1; //Fehler bei Kommunikation
228 *I2C_test_struct.status=Error;
229 if (I2C_test_struct.callback != NULL)
230 {
231 I2C_test_struct.callback(I2C_test_struct.readbuf, I2C_test_struct.num_read, I2C_test_struct.status, I2C_test_struct.ID);
232 }
233 return FSM_Stop;
234 }
235
236 if (I2C2STATbits.ACKSTAT==0) //Wenn ACK von Slave erhalten
237 {
238 static int count=0;
239
240 if (count < I2C_test_struct.num_write) //Noch Bytes zu senden
241 {
242
243 I2C2TRN=I2C_test_struct.writebuf[count];
244 count++;
245 return FSM_Adresse_Write;
246 }
247
248 else //Nichts mehr zu schicken
249 {
250 count=0;
251 I2C2CONbits.RSEN=1; // Leitet Bedingungen für den Restart weiter
252 return FSM_Repeated_Start;
253 }
254
255 }
256
257 }
258 return FSM_Adresse_Write;

260} /* *FSM_Adresse_Write() */

290void *FSM_Adresse_Read(void)
291{
292 if(I2C2STATbits.TRSTAT==0) //Wenn erfolgreich übertragen
293 {
294 if (I2C2STATbits.ACKSTAT==1) //Wenn NACK von Slave erhalten
295 {
296 I2C2CONbits.PEN=1; // Leitet Stop-Bedigungen weiter
297 *I2C_test_struct.status=Error; //Fehler bei Kommunikation
298 if (I2C_test_struct.callback != NULL)
299 {
300 I2C_test_struct.callback(I2C_test_struct.readbuf, I2C_test_struct.num_read, I2C_test_struct.status, I2C_test_struct.ID);
301 }
302 return FSM_Stop;
303 }
304
305 if (I2C2STATbits.ACKSTAT==0) //Wenn ACK von Slave erhalten
306 {
307 if (I2C2CONbits.ACKEN==0) //Wenn Bit der ACK-Sequenz freigegeben
308 {
309 static int count = 0;
310
311 if (count < I2C_test_struct.num_read) //Noch Bytes zu empfangen
312 {
313 count++;
314 I2C2CONbits.RCEN=1; // Aktiviert den Empfangsmodus für I2C
315 return FSM_RECV_EN;
316 }
317

320 count = 0;
321 I2C2CONbits.PEN=1; // Leitet Stop-Bedigungen weiter
322

324 //Callback Funktion aufrufen

331 }

352void *FSM_RECV_EN(void)

356 static int count = 0;
357 I2C_test_struct.readbuf[count]=I2C2RCV;
